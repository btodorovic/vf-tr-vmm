policy-options {
    policy-statement VRF-VFNETMNG-EXPORT {
        term EXPORT {
            from {
                protocol direct;
                route-filter $LO0-VFNETMNG-IP$/32 exact;
            }
            then {
                community add RT-VFNETMNG;
                accept;
            }
        }
        then reject;
    }
    policy-statement VRF-VFNETMNG-IMPORT {
        term IMPORT {
            from community RT-VFNETMNG;
            then accept;
        }
        then reject;
    }
    community RT-VFNETMNG members target:15924:1723;
}
